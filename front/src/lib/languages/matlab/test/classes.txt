# Simple class

classdef MyClass
  properties
    Value
  end
  methods
    function obj = MyClass(v)
      obj.Value = v
    end
  end
end

==>

Program(ClassDefinition(
  classdef, Identifier,
  PropertiesBlock(properties, Body(ExpressionStatement(Identifier)), end),
  MethodsBlock(methods,
    FunctionDeclaration(
      function, Identifier, Identifier, ParamList(Identifier),
      Body(AssignmentStatement(MemberExpression(Identifier, PropertyName), Identifier)),
      end
    ),
    end
  ),
  end
))
