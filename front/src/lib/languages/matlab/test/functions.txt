# Nested functions

function y = outer(x)
  y = inner(x)
  function z = inner(w)
    z = w + 1
  end
end

==>

Program(FunctionDeclaration(
  function, Identifier, Identifier, ParamList(Identifier),
  Body(
    AssignmentStatement(Identifier, CallExpression(Identifier, Identifier)),
    FunctionDeclaration(
      function, Identifier, Identifier, ParamList(Identifier),
      Body(AssignmentStatement(Identifier, BinaryExpression(Identifier, ArithOp, Number))),
      end
    )
  ),
  end
))

# Function with arguments block

  arguments
    x (1,1) double
  end
  y = x * 2
end

  end
